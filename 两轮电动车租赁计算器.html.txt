        // 计算核心逻辑
        function calculate() {
            const carPrice = Number(document.getElementById('carPrice').value);
            const leaseTerm = Number(document.getElementById('leaseTerm').value);
            const startDate = new Date(document.getElementById('startDate').value);
            
            // 验证输入
            if (!carPrice || !startDate.getTime()) {
                alert('请完善车辆价格和起租日期');
                return;
            }
            
            // 基础参数计算（隐藏本金、租金明细，仅用于计算总额）
            const deposit = Math.round(carPrice * 0.25);
            const monthlyPrincipal = Math.round(carPrice / leaseTerm);
            const gpsMonthly = Math.round(170 / leaseTerm);
            
            // 租金费率配置
            let rate;
            if (leaseTerm === 3 || leaseTerm === 6) rate = 0.06;
            else if (leaseTerm === 9 || leaseTerm === 12) rate = 0.05;
            else if (leaseTerm === 24) rate = 0.04;
            const monthlyRateFee = Math.round(carPrice * rate);
            
            // 每月应付总额（含本金、租金、GPS费，不拆分展示）
            const monthlyTotal = monthlyPrincipal + monthlyRateFee + gpsMonthly;
            
            // 填充基础信息
            document.getElementById('showCarPrice').textContent = carPrice + '元';
            document.getElementById('deposit').textContent = deposit + '元';
            document.getElementById('showLeaseTerm').textContent = leaseTerm + '个月';
            document.getElementById('monthlyTotal').textContent = monthlyTotal + '元';
            
            // 生成租期计划表（仅显示期数、日期、总额）
            const tableBody = document.getElementById('tableBody');
            tableBody.innerHTML = '';
            
            for (let i = 1; i <= leaseTerm; i++) {
                // 计算每期应付日期（每月同一日）
                const payDate = new Date(startDate);
                payDate.setMonth(payDate.getMonth() + i);
                const formatDate = `${payDate.getFullYear()}-${(payDate.getMonth() + 1).toString().padStart(2, '0')}-${payDate.getDate().toString().padStart(2, '0')}`;
                
                // 创建表格行
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${i}期</td>
                    <td>${formatDate}</td>
                    <td>${monthlyTotal}元</td>
                `;
                tableBody.appendChild(row);
            }
            
            // 显示结果
            document.getElementById('result').style.display = 'block';
            // 滚动到结果区域
            document.getElementById('result').scrollIntoView({ behavior: 'smooth' });
        }

        // 初始化起租日期为今日
        window.onload = function() {
            const today = new Date().toISOString().split('T')[0];
            document.getElementById('startDate').value = today;
        };
    </script>
